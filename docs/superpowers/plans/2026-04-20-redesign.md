```jsx
import { useState } from 'react'
import Logo from './components/Logo'
import Home from './pages/Home'
import RaceComparison from './pages/RaceComparison'
import Info from './pages/Info'
import TexelRatingInfo from './pages/TexelRatingInfo'
import './App.css'

const NAV = [
  { id: 'home',    label: 'Home',     Page: Home },
  { id: 'compare', label: 'Vergelijk', Page: RaceComparison },
  { id: 'schema',  label: 'Schema',   Page: Info },
  { id: 'uitleg',  label: 'Uitleg',   Page: TexelRatingInfo },
]

export default function App() {
  const [page, setPage] = useState('home')

  return (
    <div className="min-h-screen" style={{ backgroundColor: 'var(--bg)' }}>

      {/* Sticky header */}
      <header className="sticky top-0 z-50 bg-white border-b px-4 pt-3 pb-2"
        style={{ borderColor: 'var(--border)' }}>
        <Logo />
        <div className="flex mt-2 p-0.5 rounded-lg gap-0.5"
          style={{ backgroundColor: '#f3f4f6' }}>
          {NAV.map(({ id, label }) => {
            const active = page === id
            return (
              <button
                key={id}
                onClick={() => setPage(id)}
                className="flex-1 py-1.5 text-xs font-medium rounded-md transition-all"
                style={{
                  fontFamily: "'Inter', sans-serif",
                  backgroundColor: active ? '#ffffff' : 'transparent',
                  color: active ? 'var(--text)' : 'var(--text3)',
                  fontWeight: active ? 700 : 500,
                  boxShadow: active ? '0 1px 3px rgba(0,0,0,0.10)' : 'none',
                }}
              >
                {label}
              </button>
            )
          })}
        </div>
      </header>

      {/* Page content */}
      <main>
        <div className="max-w-sm mx-auto px-4 pb-8">
          {NAV.map(({ id, Page }) =>
            page === id && <Page key={id} navigate={setPage} />
          )}
        </div>
      </main>
    </div>
  )
}
```

```jsx
import { useState, useMemo } from 'react'
import boats from '../data/boats.json'

const ALL_CLASSES = ['Alle', ...new Set(boats.boats.map(b => b.class))]
const getTR = (boat, spi) => spi && boat.trWithSpi ? boat.trWithSpi : boat.trNoSpi

function SpiToggle({ value, onChange, small = false }) {
  return (
    <div className="flex gap-0.5 p-0.5 rounded-lg"
      style={{ backgroundColor: '#f3f4f6', display: 'inline-flex' }}>
      {[false, true].map(v => (
        <button key={String(v)} onClick={() => onChange(v)}
          className="rounded-md transition-all"
          style={{
            padding: small ? '3px 8px' : '5px 12px',
            fontSize: small ? '11px' : '12px',
            fontFamily: "'Inter', sans-serif",
            fontWeight: 500,
            backgroundColor: value === v ? '#fff' : 'transparent',
            color: value === v ? 'var(--text)' : 'var(--text3)',
            boxShadow: value === v ? '0 1px 2px rgba(0,0,0,0.08)' : 'none',
          }}>
          {v ? 'Met spi' : 'Geen spi'}
        </button>
      ))}
    </div>
  )
}

function BoatBrowser({ excludeIds = [], onSelect }) {
  const [activeClass, setActiveClass] = useState('Alle')
  const [search, setSearch] = useState('')

  const filtered = useMemo(() =>
    boats.boats
      .filter(b =>
        !excludeIds.includes(b.id) &&
        (activeClass === 'Alle' || b.class === activeClass) &&
        b.type.toLowerCase().includes(search.toLowerCase())
      )
      .sort((a, b) => a.type.localeCompare(b.type))
  , [activeClass, search, excludeIds])

  return (
    <div>
      <input
        type="text"
        placeholder="Zoek boot…"
        value={search}
        onChange={e => setSearch(e.target.value)}
        className="w-full px-3.5 py-2.5 rounded-xl text-sm mb-2.5 focus:outline-none"
        style={{
          fontFamily: "'Inter', sans-serif",
          border: '1.5px solid var(--border)',
          backgroundColor: 'var(--surface)',
          color: 'var(--text)',
        }}
      />
      <div className="flex gap-1.5 flex-wrap mb-3">
        {ALL_CLASSES.map(cls => (
          <button key={cls} onClick={() => setActiveClass(cls)}
            className="px-3 py-1 rounded-full text-xs font-medium transition-all"
            style={{
              fontFamily: "'Inter', sans-serif",
              backgroundColor: activeClass === cls ? 'var(--accent)' : 'var(--surface)',
              color: activeClass === cls ? '#fff' : 'var(--text2)',
              border: `1px solid ${activeClass === cls ? 'var(--accent)' : 'var(--border)'}`,
            }}>
            {cls}
          </button>
        ))}
      </div>
      <div className="card overflow-hidden" style={{ maxHeight: '220px', overflowY: 'auto' }}>
        {filtered.length === 0 && (
          <p className="px-4 py-3 text-sm text-center" style={{ color: 'var(--text3)', fontFamily: "'Inter', sans-serif" }}>
            Geen boten gevonden
          </p>
        )}
        {filtered.map(b => (
          <button key={b.id} onClick={() => onSelect(b)}
            className="w-full flex items-center justify-between px-4 py-2.5 text-left transition-colors hover:bg-gray-50"
            style={{ borderBottom: '1px solid var(--border)' }}>
            <div>
              <p className="text-sm font-semibold" style={{ fontFamily: "'Plus Jakarta Sans', sans-serif", color: 'var(--text)' }}>
                {b.type}
              </p>
              <p className="text-xs" style={{ fontFamily: "'Inter', sans-serif", color: 'var(--text3)' }}>
                {b.class} · {b.crew === 1 ? '1 persoon' : `${b.crew} personen`}
              </p>
            </div>
            <span className="text-sm font-medium ml-2 flex-shrink-0"
              style={{ fontFamily: "'DM Mono', monospace", color: 'var(--accent)' }}>
              {b.trNoSpi}{b.trWithSpi ? `/${b.trWithSpi}` : ''}
            </span>
          </button>
        ))}
      </div>
    </div>
  )
}

function fmtDiff(seconds) {
  const abs = Math.abs(Math.round(seconds / 60))
  const h = Math.floor(abs / 60), m = abs % 60
  if (h > 0) return `${h}u ${m}m`
  return `${m} min`
}

export default function RaceComparison() {
  const [selectedBoat, setSelectedBoat] = useState(null)
  const [ownSpi, setOwnSpi] = useState(false)
  const [finishTime, setFinishTime] = useState('04:00')
  const [compBoats, setCompBoats] = useState([])
  const [addingBoat, setAddingBoat] = useState(false)

  const finishSec = useMemo(() => {
    const [h, m] = finishTime.split(':').map(Number)
    return (h * 60 + (m || 0)) * 60
  }, [finishTime])

  const ownTR = selectedBoat ? getTR(selectedBoat, ownSpi) : null
  const excludeIds = [selectedBoat?.id, ...compBoats.map(b => b.boat.id)].filter(Boolean)

  const scoreboard = useMemo(() => {
    if (!selectedBoat || !finishSec || !ownTR) return []
    return compBoats
      .map(({ boat, spi }) => {
        const compTR = getTR(boat, spi)
        const allowed = finishSec * (compTR / ownTR)
        const diff = allowed - finishSec
        return { boat, spi, compTR, diff }
      })
      .sort((a, b) => b.diff - a.diff)
  }, [selectedBoat, finishSec, compBoats, ownTR])

  function handleSelectBoat(boat) {
    setSelectedBoat(boat)
    setCompBoats([])
    setAddingBoat(false)
  }

  return (
    <div>
      {/* Sectie 1: Eigen boot */}
      <p className="section-title">Jouw boot</p>

      {!selectedBoat ? (
        <BoatBrowser excludeIds={[]} onSelect={handleSelectBoat} />
      ) : (
        <div className="card px-4 py-3"
          style={{ border: '1.5px solid var(--accent)', backgroundColor: 'var(--accent-bg)' }}>
          <div className="flex items-start justify-between mb-2">
            <div>
              <p className="text-base font-bold"
                style={{ fontFamily: "'Plus Jakarta Sans', sans-serif", color: 'var(--text)' }}>
                {selectedBoat.type}
              </p>
              <p className="text-xs mt-0.5"
                style={{ fontFamily: "'Inter', sans-serif", color: 'var(--text2)' }}>
                {selectedBoat.class} · TR{' '}
                <span style={{ fontFamily: "'DM Mono', monospace", fontWeight: 500, color: 'var(--accent)' }}>
                  {ownTR}
                </span>
              </p>
            </div>
            <button
              onClick={() => { setSelectedBoat(null); setCompBoats([]) }}
              className="text-xs font-semibold px-3 py-1.5 rounded-lg"
              style={{ backgroundColor: 'var(--accent)', color: '#fff', fontFamily: "'Inter', sans-serif" }}>
              Wijzig
            </button>
          </div>
          <SpiToggle value={ownSpi} onChange={setOwnSpi} />
        </div>
      )}

      {/* Sectie 2: Tijd */}
      {selectedBoat && (
        <>
          <p className="section-title">Jouw tijd (uu:mm)</p>
          <div className="flex gap-2 items-center">
            <input
              type="time"
              value={finishTime}
              onChange={e => setFinishTime(e.target.value)}
              className="flex-1 px-4 py-3 rounded-xl text-lg text-center focus:outline-none"
              style={{
                fontFamily: "'DM Mono', monospace",
                fontWeight: 500,
                border: '1.5px solid var(--border)',
                backgroundColor: 'var(--surface)',
                color: 'var(--text)',
              }}
            />
            <button
              onClick={() => setAddingBoat(true)}
              className="px-4 py-3 rounded-xl text-sm font-semibold flex-shrink-0"
              style={{
                fontFamily: "'Inter', sans-serif",
                backgroundColor: 'var(--accent)',
                color: '#fff',
              }}>
              + Boot
            </button>
          </div>
        </>
      )}

      {/* Boot toevoegen browser */}
      {addingBoat && (
        <>
          <p className="section-title">Boot toevoegen</p>
          <BoatBrowser
            excludeIds={excludeIds}
            onSelect={b => {
              setCompBoats(p => [...p, { boat: b, spi: false }])
              setAddingBoat(false)
            }}
          />
        </>
      )}

      {/* Sectie 3: Scoreboard */}
      {selectedBoat && (
        <>
          <p className="section-title">Scoreboard</p>
          {scoreboard.length === 0 && (
            <p className="text-sm text-center py-6"
              style={{ fontFamily: "'Inter', sans-serif", color: 'var(--text3)' }}>
              Voeg een boot toe om te vergelijken
            </p>
          )}
          <div className="flex flex-col gap-2.5">
            {scoreboard.map(({ boat, spi, compTR, diff }) => (
              <div key={boat.id} className="card px-4 py-3 flex items-center justify-between"
                style={{
                  borderLeft: `3px solid ${diff >= 0 ? 'var(--green)' : 'var(--red)'}`,
                }}>
                <div className="flex-1 min-w-0 mr-3">
                  <p className="text-sm font-bold truncate"
                    style={{ fontFamily: "'Plus Jakarta Sans', sans-serif", color: 'var(--text)' }}>
                    {boat.type}
                  </p>
                  <div className="flex items-center gap-2 mt-1">
                    <span className="text-xs" style={{ fontFamily: "'Inter', sans-serif", color: 'var(--text3)' }}>
                      TR <span style={{ fontFamily: "'DM Mono', monospace" }}>{compTR}</span>
                    </span>
                    <SpiToggle value={spi} small
                      onChange={() => setCompBoats(p =>
                        p.map(b => b.boat.id === boat.id ? { ...b, spi: !b.spi } : b)
                      )} />
                  </div>
                </div>
                <div className="flex items-center gap-2 flex-shrink-0">
                  <div className="text-right">
                    <p className="text-xl font-bold leading-none"
                      style={{
                        fontFamily: "'DM Mono', monospace",
                        color: diff >= 0 ? 'var(--green)' : 'var(--red)',
                      }}>
                      {diff >= 0 ? '+' : '−'}{fmtDiff(diff)}
                    </p>
                    <p className="text-xs mt-0.5"
                      style={{ fontFamily: "'Inter', sans-serif", color: 'var(--text3)' }}>
                      {diff >= 0 ? 'voordeel' : 'nadeel'}
                    </p>
                  </div>
                  <button
                    onClick={() => setCompBoats(p => p.filter(b => b.boat.id !== boat.id))}
                    className="w-7 h-7 rounded-full flex items-center justify-center text-base leading-none flex-shrink-0"
                    style={{ backgroundColor: '#f3f4f6', color: 'var(--text3)' }}>
                    ×
                  </button>
                </div>
              </div>
            ))}
          </div>
        </>
      )}
      <div className="pb-4" />
    </div>
  )
}
```
